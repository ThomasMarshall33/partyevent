    <script>
        let events = [];

        function formatDate(dateStr) {
            const date = new Date(dateStr);
            return date.toLocaleDateString('en-US', { 
                weekday: 'long', 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric' 
            });
        }

        function formatTime(timeStr) {
            const [hours, minutes] = timeStr.split(':');
            const date = new Date();
            date.setHours(hours, minutes);
            return date.toLocaleTimeString('en-US', { 
                hour: 'numeric', 
                minute: '2-digit',
                hour12: true 
            });
        }

        function showSuccessMessage() {
            const message = document.getElementById('successMessage');
            message.style.display = 'block';
            setTimeout(() => {
                message.style.display = 'none';
            }, 4000);
        }

        function renderEvents() {
            const container = document.getElementById('eventsContainer');
            
            if (events.length === 0) {
                container.innerHTML = '<div class="no-events">No parties posted yet. Be the first to get this party started! üéâ</div>';
                return;
            }

            // Sort events by date and time
            const sortedEvents = events.sort((a, b) => {
                const dateA = new Date(a.date + ' ' + a.time);
                const dateB = new Date(b.date + ' ' + b.time);
                return dateA - dateB;
            });

            container.innerHTML = sortedEvents.map((event, index) => `
                <div class="event-card">
                    <div class="event-title">${event.title}</div>
                    <div class="event-meta">
                        <span>üìÖ ${formatDate(event.date)}</span>
                        <span>üïí ${formatTime(event.time)}</span>
                        <span>üìç ${event.location}</span>
                        ${event.price ? `<span>üí∞ ${event.price}</span>` : ''}
                        ${event.organizer ? `<span>üéØ ${event.organizer}</span>` : ''}
                    </div>
                    <div class="event-description">${event.description}</div>
                    <div class="event-footer">
                        <div class="event-category">${event.category}</div>
                        <button class="delete-btn" onclick="deleteEvent(${index})">Remove</button>
                    </div>
                </div>
            `).join('');
        }

        function deleteEvent(index) {
            if (confirm('Are you sure you want to remove this event?')) {
                events.splice(index, 1);
                renderEvents();
            }
        }

        // Set minimum date to today
        document.getElementById('eventDate').min = new Date().toISOString().split('T')[0];
    </script>
